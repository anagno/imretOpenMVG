0. Install dependencies

sudo apt-get install git cmake cmake-gui

1. Install cuda
http://docs.nvidia.com/cuda/cuda-getting-started-guide-for-linux/index.html#ubuntu-installation

2. Install openmvg

sudo apt-get install libpng-dev libjpeg-dev libtiff-dev libxxf86vm1 libxxf86vm-dev libxi-dev libxrandr-dev 
git clone --recursive https://github.com/openMVG/openMVG.git
mkdir openMVG_Build
cd openMVG_Build
cmake -DCMAKE_BUILD_TYPE=RELEASE . ../openMVG/src/
make -j4
sudo make install

3. Install flann (https://github.com/mariusmuja/flann.git)
sudo apt-get install libflann-dev libflann1.8

4. Install qt5
sudo apt-get install qt5-default

5. Install libconfig++
sudo apt-get install libconfig++-dev

6. Install more libraries
sudo apt-get install libboost-all-dev

7. Install the latest cmake version
sudo apt-get install software-properties-common
sudo add-apt-repository ppa:george-edison55/cmake-3.x
sudo apt-get update
sudo apt-get install cmake

8. Install SiftGPU (https://github.com/pitzer/SiftGPU)
sudo apt-get install libglew-dev libglew1.10 glew-utils
cd
donwload and unzip it from http://www.cs.unc.edu/~ccwu/siftgpu/download.html
cd SiftGPU
make
sudo cp bin/libsiftgpu.so /usr/lib

9. Compile the program
cd
git clone https://github.com/anagno/imretOpenMVG.git
cd imretOpenMVG/
mkdir build
cd build
cmake . ../src/
